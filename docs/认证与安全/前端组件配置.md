# 滑块验证码组件配置

## 前端组件选择

### 推荐组件库

#### 1. [vue3-slide-verify](https://github.com/Justinium/vite-plugin-pinyin) (Vue3)

```bash
npm install vue3-slide-verify
```

```vue
<template>
  <Vue3SlideVerify
    ref="sliderRef"
    title="请按住滑块完成验证"
    text="向右滑动完成验证"
    width="310"
    height="155"
    :bar-size="40"
    :mode="mode"
    :img-src="captchaData.sliderImage"
    :backdrop-src="captchaData.backgroundImage"
    @success="onSuccess"
    @fail="onFail"
    @retry="onRetry"
  />
</template>

<script setup>
import { ref } from 'vue'
import { Vue3SlideVerify } from 'vue3-slide-verify'

const sliderRef = ref()
const captchaData = ref(null)

// 获取验证码
const getCaptcha = async () => {
  const res = await fetch('/api/v1/auth/captcha/slider')
  captchaData.value = await res.json()
}

// 验证成功
const onSuccess = async (data) => {
  console.log('验证成功', data)
  // data 包含 captchaId, x, t, track 等信息
  // 发送到后端验证
  const res = await fetch('/api/v1/auth/captcha/slider/verify', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      captchaId: data.captchaId,
      captchaResult: JSON.stringify({
        x: data.x,
        t: data.t,
        track: data.track
      })
    })
  })
  const result = await res.json()
  if (result.valid) {
    console.log('验证通过')
  }
}

// 验证失败
const onFail = (data) => {
  console.log('验证失败', data)
  // 重新获取验证码
  getCaptcha()
}

// 重试
const onRetry = () => {
  getCaptcha()
}

// 初始化
getCaptcha()
</script>
```

### 自定义组件 (Canvas 实现)

如果需要完全自定义，可以使用 Canvas 实现：

```vue
<template>
  <div class="slider-captcha">
    <div class="captcha-container" ref="containerRef" @mousemove="onMouseMove" @mouseup="onMouseUp">
      <canvas ref="canvasRef" width="310" height="155"></canvas>
      <div class="slider" :style="{ left: sliderLeft + 'px' }" @mousedown="onMouseDown">
        <span>拖动滑块</span>
      </div>
    </div>
    <p class="captcha-tips">{{ tips }}</p>
  </div>
</template>

<script setup>
import { ref, onMounted } from 'vue'

const containerRef = ref(null)
const canvasRef = ref(null)
const sliderLeft = ref(0)
const isDragging = ref(false)
const startPos = ref(0)
const tips = ref('请按住滑块完成验证')

// 绘制验证码
const drawCaptcha = async () => {
  const canvas = canvasRef.value
  const ctx = canvas.getContext('2d')
  
  // 绘制背景图
  const backgroundImg = new Image()
  backgroundImg.src = captchaData.value.backgroundImage
  await new Promise(resolve => {
    backgroundImg.onload = resolve
    backgroundImg.onerror = resolve
  })
  ctx.drawImage(backgroundImg, 0, 0, 310, 155)
  
  // 绘制缺口
  const x = captchaData.value.correctX
  const y = Math.floor(Math.random() * (155 - 40))
  ctx.fillStyle = '#fff'
  ctx.beginPath()
  ctx.arc(x + 20, y + 20, 20, 0, Math.PI * 2)
  ctx.fill()
  
  // 绘制滑块
  const sliderImg = new Image()
  sliderImg.src = captchaData.value.sliderImage
  await new Promise(resolve => {
    sliderImg.onload = resolve
    sliderImg.onerror = resolve
  })
  ctx.drawImage(sliderImg, x, y, 40, 40)
}

// 鼠标按下
const onMouseDown = (e) => {
  isDragging.value = true
  startPos.value = e.clientX
  tips.value = '按住滑块拖动'
}

// 鼠标移动
const onMouseMove = (e) => {
  if (!isDragging.value) return
  
  const deltaX = e.clientX - startPos.value
  const maxLeft = 310 - 40
  
  sliderLeft.value = Math.max(0, Math.min(deltaX, maxLeft))
}

// 鼠标松开
const onMouseUp = async () => {
  if (!isDragging.value) return
  isDragging.value = false
  
  const currentX = sliderLeft.value
  const t = Date.now() - startTime.value
  
  // 验证
  const res = await fetch('/api/v1/auth/captcha/slider/verify', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      captchaId: captchaData.value.captchaId,
      captchaResult: JSON.stringify({
        x: currentX,
        t: t,
        track: [] // 可以记录轨迹
      })
    })
  })
  const result = await res.json()
  
  if (result.valid) {
    tips.value = '验证通过'
    // 执行登录逻辑
  } else {
    tips.value = '验证失败，请重试'
    // 重新获取验证码
    await getCaptcha()
  }
}
</script>

<style scoped>
.slider-captcha {
  width: 310px;
  margin: 0 auto;
}

.captcha-container {
  position: relative;
  width: 310px;
  height: 155px;

```

## API 接口说明

### 1. 获取验证码

```
GET /api/v1/auth/captcha/slider
```

**响应示例：**
```json
{
  "captchaId": "uuid",
  "backgroundImage": "/api/v1/auth/captcha/image/xxx/background",
  "sliderImage": "/api/v1/auth/captcha/image/xxx/slider",
  "correctX": 0,
  "width": 310,
  "height": 155,
  "blockSize": 40
}
```

### 2. 验证验证码

```
POST /api/v1/auth/captcha/slider/verify
Content-Type: application/json

{
  "captchaId": "uuid",
  "captchaResult": "{\"x\": 100, \"t\": 1500, \"track\": [99, 99, 100]}"
}
```

**响应示例：**
```json
{
  "valid": true,
  "message": "验证通过"
}
```

## 系统配置

在 `CfgSystemConfig` 表中配置：

| 配置键 | 说明 | 默认值 |
|--------|------|--------|
| `login_captcha_enabled` | 登录验证码开关 | 1 |
| `captcha_type` | 验证码类型 (slider/image) | slider |
| `captcha_width` | 验证码宽度 | 310 |
| `captcha_height` | 验证码高度 | 155 |
| `captcha_block_size` | 滑块大小 | 40 |

## 使用方式

1. **安装前端依赖**
```bash
npm install vue3-slide-verify
# 或
npm install ant-design-vue
```

2. **在登录页面引入组件**
```vue
<template>
  <div v-if="captchaEnabled">
    <Vue3SlideVerify
      ref="sliderRef"
      @success="onCaptchaSuccess"
      @fail="onCaptchaFail"
    />
  </div>
</template>

<script setup>
import { ref } from 'vue'
import { Vue3SlideVerify } from 'vue3-slide-verify'

const captchaEnabled = ref(true)
const sliderRef = ref()
const captchaData = ref(null)

const getCaptcha = async () => {
  const res = await fetch('/api/v1/auth/captcha/slider')
  captchaData.value = await res.json()
}

const onCaptchaSuccess = async (data) => {
  // 验证通过，执行登录
  const res = await fetch('/api/v1/auth/captcha/slider/verify', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      captchaId: data.captchaId,
      captchaResult: JSON.stringify(data)
    })
  })
  const result = await res.json()
  if (result.valid) {
    // 执行登录
    await login()
  }
}

const onCaptchaFail = () => {
  getCaptcha() // 重新获取验证码
}

getCaptcha()
</script>
```
